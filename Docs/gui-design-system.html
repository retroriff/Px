  var sig, env;
  env = EnvGen.kr(Env.adsr(0.1, 0.3, 0.7, 1), gate, doneAction: 2);
  sig = LFSaw.ar(freq * [1, 1.01]);
  sig = LPF.ar(sig, 2000);
  Out.ar(0, sig * env * amp);
}).add;
